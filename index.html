    const STORAGE_KEY = "smileWife_playData_v1";

    const SUPPORT_KEY = "smileWife_supportLog_v1";


    function getTodayString() {

      const d = new Date();

      const y = d.getFullYear();

      const m = String(d.getMonth() + 1).padStart(2, "0");

      const day = String(d.getDate()).padStart(2, "0");

      return `${y}-${m}-${day}`;

    }


    function loadPlayData() {

      const today = getTodayString();

      let data;

      try {

        data = JSON.parse(localStorage.getItem(STORAGE_KEY));

      } catch (e) {

        data = null;

      }

      if (!data || data.date !== today) {

        data = {

          date: today,

          playCount: 0,

          unlimited: false

        };

        localStorage.setItem(STORAGE_KEY, JSON.stringify(data));

      }

      return data;

    }


    function savePlayData(data) {

      localStorage.setItem(STORAGE_KEY, JSON.stringify(data));

    }


    function updateStatusUI() {

      const data = loadPlayData();

      const today = getTodayString();

      const todayText = document.getElementById("todayText");

      const playText = data.unlimited

        ? `今日のプレイ：${data.playCount}回（無制限パス中）`

        : `今日のプレイ：${data.playCount} / ${freeLimit}回（無料）`;


      playStatus.textContent = playText;

      passStatus.textContent = data.unlimited ? "無制限パス：有効" : "無制限パス：なし";

    }
